module top_module (
    input clk,
    input reset,            // Synchronous reset
    input [7:0] d,
    output [7:0]  q
);
    
    reg [7:0] q_reg;
    // always@(posedge clk or posedge reset) 
    always@(posedge clk)
    begin
        if (reset==1)
            q_reg <= 7'd0;
        else  
        	q_reg <= d;
     
    end
    
    assign q = q_reg;
    
    
    

    
    
    

endmodule
