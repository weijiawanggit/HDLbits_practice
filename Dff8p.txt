module top_module (
    input clk,
    input reset,
    input [7:0] d,
    output [7:0] q
);

     reg [7:0] q_reg;
    // always@(posedge clk or posedge reset) 
    always@(negedge clk)
    begin
        if (reset==1)
            q_reg <= 7'h0x34;
        else  
        	q_reg <= d;
     
    end
    
    assign q = q_reg;
    

    
    
    
    
endmodule
